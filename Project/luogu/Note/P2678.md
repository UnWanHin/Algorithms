    #include <iostream>
    #include <algorithm>
    #include <vector>
    using namespace std;
    
    int SelectRefines(vector<int>& v, int left, int right);
    
    int main() {
    int L,N,M;
    cin>>L>>N>>M;
    if (L<1||N<0||M<0||M>N) {
    return -1;
    }

    vector<int> Rock(N,0);
    for (int i = 0; i < N; i++) {
        cin>>Rock[i];
    }
    //sorted and not repeatable default
    for (int i=0; i<M; i++) {
    int minimum = Rock[0];
    int position = 0;
    for (int j=1;j<Rock.size();j++) {
    if (Rock[j]-Rock[j-1]<minimum) {
    minimum = Rock[j]-Rock[j-1];
    position = SelectRefines(Rock,j-1,j);
    }
    }
    //last judge 不用判斷左右了因為是邊界
    if (L-Rock[N-1]<minimum) {
    minimum = L-Rock[N-1];
    position = N-1;
    }
    Rock.erase(Rock.begin()+position);
    }

    for (auto& i: Rock) {
        cout<<i<<" ";
    }
    int minimum = 0;
    for (int i=1;i<Rock.size();i++) {
        int position = 0;
        minimum = Rock.at(0);
        if (Rock[i]-Rock[i-1]<minimum) {
            minimum = Rock[i]-Rock[i-1];
            position = i;
        }
    }
    cout<<"Minimum : "<<endl;
    cout<<minimum<<endl;

    return 0;
    }
    
    int SelectRefines(vector<int>& v, int left, int right) {
    int position=-1;
    if (v.at(left)-v.at(left-1)==v.at(right+1)-v.at(right)) {
    position = left;
    }
    if (v.at(left)-v.at(left-1)==v.at(right+1)-v.at(right)) {
    position = right;
    }
    return position;
    }
